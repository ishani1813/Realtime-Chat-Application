app.use(express.json());

// Room registry: Map<roomId, Set<WebSocket>>
const rooms = new Map();

wss.on('connection', (ws, req) => {
  const { room, userId } = parseQuery(req.url);

  // Add socket to room
  if (!rooms.has(room)) rooms.set(room, new Set());
  rooms.get(room).add(ws);

  ws.on('message', async (raw) => {
    const msg = JSON.parse(raw);

    // Broadcast to all peers in room (non-blocking)
    broadcast(rooms.get(room), msg, ws);

    // Async DB write — does not block broadcast
    await persistMessage({ ...msg, room, userId });
  });

  ws.on('close', () => {
    rooms.get(room)?.delete(ws);
  });
});

useChat.js React Hook 
import { useState, useEffect, useRef, useCallback }
  from 'react';

/**
 * Custom hook — manages WebSocket lifecycle
 * and message state for a given room.
 */
export function useChat(room) {
  const [messages, setMessages] = useState([]);
  const [connected, setConnected] = useState(false);
  const [typingUsers, setTypingUsers] = useState([]);
  const wsRef = useRef(null);

  useEffect(() => {
    const ws = new WebSocket(
      `ws://localhost:8080?room=${room}`
    );

    ws.onopen  = () => setConnected(true);
    ws.onclose = () => setConnected(false);

    ws.onmessage = ({ data }) => {
      const event = JSON.parse(data);

      if (event.type === 'message') {
        setMessages(prev => [...prev, event]);
      } else if (event.type === 'typing') {
        setTypingUsers(event.users);
      }
    };

    wsRef.current = ws;
    return () => ws.close();
  }, [room]);

  const send = useCallback((text) => {
    wsRef.current?.send(
      JSON.stringify({ type: 'message', text })
    );
  }, []);

  return { messages, send, connected, typingUsers };
}
broadcast.js Utility 
const WebSocket = require('ws');

/**
 * Broadcast a message to all open sockets
 * in a room, optionally excluding sender.
 */
function broadcast(roomClients, payload, sender = null) {
  if (!roomClients) return;

  const data = JSON.stringify(payload);

  roomClients.forEach((client) => {
    if (
      client !== sender &&
      client.readyState === WebSocket.OPEN
    ) {
      client.send(data);
    }
  });
}

/**
 * Send a typing event to room peers.
 * Debounced on the client side.
 */
function broadcastTyping(roomClients, userId, sender) {
  broadcast(
    roomClients,
    { type: 'typing', userId },
    sender
  );
}

/**
 * Get count of active connections in room.
 */
function getRoomSize(roomClients) {
  if (!roomClients) return 0;
  return [...roomClients].filter(
    c => c.readyState === WebSocket.OPEN
  ).length;
}

module.exports = { broadcast, broadcastTyping, getRoomSize };
